217void RobotG7::obtenirVitesseJoystick()
218{
219 float xval = GamePad.getXaxisData();
220 float yval = GamePad.getYaxisData();
221
222 //Pour la calibration
223 if (xval < -6) {
224 xval = -6;
225 this->gamePadGestionCommand();
226 } else if (xval > 6) {
227 xval = 6;
228 this->gamePadGestionCommand();
229 }
230 if (yval < -6) {
231 yval = -6;this->gamePadGestionCommand();
232 } else if (yval > 6) {
233 yval = 6;this->gamePadGestionCommand();
234 }
235
236 int yMapped = 0;
237 if (yval < 0 )
238 {
239 this->deplacementArriere();this->gamePadGestionCommand();
240 yMapped = map(yval, -6, 0, 250, 0);
241 }
242 else
243 {
244 this->deplacementAvant();this->gamePadGestionCommand();
245 yMapped = map(yval, 0, 6, 0, 250);
246 }
247
248 int xMapped;
249 if (xval < 0 ){
250 xMapped = map(xval, -6, 0, 250, 0);this->gamePadGestionCommand();
251 }
252 else {
253 xMapped = map(xval, 0, 6, 0, 250);this->gamePadGestionCommand();
254 }
255
256 if (xval < 0 ) {
257 this->m_vitesseGauche = yMapped + xMapped;this->gamePadGestionCommand();
258 if (this->m_vitesseGauche > 250) {
259 this->m_vitesseGauche = 250;this->gamePadGestionCommand();
260 }
261
262 this->m_vitesseDroite = yMapped - xMapped;this->gamePadGestionCommand();
263 if (this->m_vitesseDroite < 0) {this->gamePadGestionCommand();
264 this->m_vitesseDroite = 0;
265 }
266
267 } else {
268 this->m_vitesseGauche = yMapped - xMapped;this->gamePadGestionCommand();
269 if (this->m_vitesseGauche < 0) {this->gamePadGestionCommand();
270 this->m_vitesseGauche = 0;
271 }// pwmchannel
272
273 this->m_vitesseDroite = yMapped + xMapped;this->gamePadGestionCommand();
274 if (this->m_vitesseDroite > 250) {this->gamePadGestionCommand();
275 this->m_vitesseDroite = 250;
276 }
277 }
